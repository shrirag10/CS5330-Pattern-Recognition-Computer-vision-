\section*{Project Overview}
% ============================================================

This project explores fundamental image processing techniques using C++ and
OpenCV. We built an interactive program that captures live video from a
webcam and applies a suite of real-time filters activated by keypresses.
Filters include greyscale conversion (standard and custom-weighted), sepia
tone, Gaussian blur (two implementations with a speed comparison), Sobel
edge detection in both the X and Y directions, gradient magnitude
visualization, blur-and-quantize posterization, Haar-cascade face detection,
monocular depth estimation using the Depth Anything V2 neural network, and
three custom effects (color negative, emboss, and depth-based exponential
fog). All pixel-level manipulation functions are implemented from scratch in
\texttt{filter.cpp} without using OpenCV filter functions where prohibited.
The project deepened our understanding of the spatial structure of images,
the efficiency trade-offs of different implementation strategies, and the
power of combining classical CV techniques with modern deep networks.

    $0.15R + 0.35G + 0.50B$, weighting blue $4.4\times$ more than the
    ITU standard. This inverts the tonal hierarchy: warm (red/orange)
    regions appear darker and cool (blue/sky) regions appear brighter
    compared to the standard output. All three output channels are set
    to the same computed value so the image is truly greyscale.
    Pixels are accessed directly via row pointers (\texttt{ptr<Vec3b>})
    for efficiency.}

        \includegraphics[width=\textwidth]{report_imgs/img3_sepia.png}
        \caption{Sepia tone output}
    \end{subfigure}
    \caption{Task 5 --- Sepia tone filter using the standard antique-camera
    matrix. Each new channel is computed as a weighted sum of all three
    \emph{original} BGR values:
    $B' = 0.272R + 0.534G + 0.131B$,
    $G' = 0.349R + 0.686G + 0.168B$,
    $R' = 0.393R + 0.769G + 0.189B$.
    To ensure original values are used throughout, all three channels are
    read into separate \texttt{float} variables \emph{before} any output
    is written. This prevents a modified channel from contaminating
    subsequent calculations. Output values are clamped to $[0, 255]$
    because the red and green coefficient rows sum to more than 1.}
\end{figure}

\subsubsection*{Timing Comparison (960$\times$720 image, avg.\ 5 runs)}

\begin{table}[H]
\centering
\begin{tabular}{lll}
\toprule
\textbf{Function} & \textbf{Avg.\ time} & \textbf{Approach} \\
\midrule
\texttt{blur5x5\_1} & 16.01\,ms & Naive nested loop with \texttt{.at<>} \\
\texttt{blur5x5\_2} & 3.81\,ms  & Separable passes + row pointers \\
\bottomrule
\end{tabular}
\caption{$\approx 4.2\times$ speedup from two changes:
(1)~\textbf{Separable filtering} --- the 2D kernel is split into a
horizontal $[1\;2\;4\;2\;1]$ pass then a vertical $[1\;2\;4\;2\;1]$ pass,
reducing operations per pixel from 25 to 10.
(2)~\textbf{Row pointer access} --- \texttt{ptr<Vec3b>()} avoids the
bounds-checking overhead of \texttt{.at<>()}.}
\end{table}

    \caption{Tasks 7--8 --- Sobel edge detection and gradient magnitude.
    \texttt{sobelX3x3()} detects \emph{vertical} edges (bright where intensity
    changes left-to-right); positive values indicate a right-going gradient.
    It decomposes as horizontal $[-1\;0\;1]$ then vertical $[1\;2\;1]$.
    \texttt{sobelY3x3()} detects \emph{horizontal} edges (positive = upward
    gradient), decomposed as horizontal $[1\;2\;1]$ then vertical $[1\;0\;-1]$.
    Both output \texttt{CV\_16SC3} (signed short) since values span
    $[-1020, 1020]$. \texttt{cv::convertScaleAbs} is used only for
    visualization; the raw signed output is kept as a separate variable.
    The magnitude image combines both: $M = \sqrt{s_x^2 + s_y^2}$
    per channel, clamped to $[0,255]$.}
\end{figure}

\begin{figure}[H]
    \centering
    \begin{subfigure}[b]{0.48\textwidth}
        \includegraphics[width=\textwidth]{report_imgs/00_original.png}
        \caption{Original color frame}
    \end{subfigure}
    \hfill
    \begin{subfigure}[b]{0.48\textwidth}
        \includegraphics[width=\textwidth]{report_imgs/img6_blurquantize.png}
        \caption{After \texttt{blurQuantize} (10 levels)}
    \end{subfigure}
    \caption{Task 9 --- Blur and quantize (poster/cartoon effect).
    \texttt{blurQuantize()} first applies \texttt{blur5x5\_2}, then
    quantizes each channel into \texttt{levels} discrete values:
    $\text{bucket} = 255 / \text{levels}$,
    $x_t = \lfloor x / \text{bucket} \rfloor$,
    $x_f = x_t \times \text{bucket}$.
    With 10 levels, each channel can take 10 values,
    yielding at most $10^3 = 1{,}000$ possible colors. The blur first
    smooths fine texture so quantization produces clean, flat color bands
    rather than noisy patches.}
\end{figure}

\subsection*{Required Images 10--12 --- Custom Effects (Task 12)}
%------------------------------------------------------------

% PARTNER SECTION --- insert negative, emboss, fog screenshots
\textit{[Images to be provided by Shyam Srinivasan --- press \texttt{n},
\texttt{e}, \texttt{o} respectively.]}

\begin{description}
    \item[Color Negative (\texttt{n}):] Pixel-wise inversion of each channel:
        $\text{dst}[c] = 255 - \text{src}[c]$. No neighborhood access required.
    \item[Emboss (\texttt{e}):] Uses the signed Sobel X and Y outputs from
        Tasks 7--8 and takes a dot product with light direction
        $(0.7071,\,0.7071)$, averaged across channels and shifted by $+128$.
        Flat regions map to mid-grey; edges facing the light appear bright.
    \item[Depth Fog (\texttt{o}):] Exponential fog using DA2 depth values.
        Distance $= 1 - d/255$; fog amount $= 1 - e^{-k\,\cdot\,\text{distance}}$;
        final color blends source with grey $(200,200,200)$.
        Close objects stay clear; far objects fade to fog.
\end{description}
